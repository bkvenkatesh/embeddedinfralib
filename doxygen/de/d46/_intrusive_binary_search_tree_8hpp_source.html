427

429  {

431  return false;
432  if (!compare(*node, *node->right))
433  return false;
434  }
435 
436  // Only the front node has no previousInOrder
437  if ((node->previousInOrder == nullptr) != (node == frontNode))
438  return false;
439  // Only the back node has no nextInOrder
440  if ((node->nextInOrder == nullptr) != (node == backNode))
441  return false;
442

444  if (nextInOrder->right != nullptr)
445  {

447

449  nextInOrder = nextInOrder->left;
450  }
451  else

458  else
