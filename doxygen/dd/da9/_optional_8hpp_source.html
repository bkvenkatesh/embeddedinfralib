4 // Optional<T> is a class that either holds an object of type T, or it does not. With
5 // this class, you can delay construction of an object, or hasten destruction. The storage
6 // needed is allocated inside of the optional. Basically, sizeof(Optional<T>) == sizeof(T) + 1.
7 // It uses pointer-like operations for access. Examples:
8 //
9 // Optional<MyClass> opt(inPlace, 2, 3, 4); // 2, 3, 4 is forwarded to the constructor of MyClass
10 //
11 // if (opt) // Test whether opt contains a value
12 // opt->MyFunc(); // Use * and -> to access the value
13 // opt = nullptr; // Reset the optional by assigning nullptr
14 // if (!opt)
15 // opt.Emplace(2, 3, 4); // Late construction by explicitly calling Emplace
16 // opt = MyClass(2, 3, 4); // Use the copy constructor for construction
