78  void erase_heap(RandomAccessIterator begin, RandomAccessIterator end, RandomAccessIterator position, Compare compare)
79  {
80  std::size_t size = std::distance(begin, end);
81  if (size > 1)
82  {
83  while (true)
84  {
85  size_t index = std::distance(begin, position);
86  RandomAccessIterator parent = index > 0 ? begin + (index - 1) / 2 : end;
87  RandomAccessIterator child1 = index * 2 + 1 < size - 1 ? begin + index * 2 + 1 : end;
88  RandomAccessIterator child2 = index * 2 + 2 < size - 1 ? begin + index * 2 + 2 : end;
89 
90  if (parent != end && compare(*parent, *std::prev(end)))
91  {
92  *position = std::move(*parent);
93  position = parent;
94  }
95  else if (child1 != end && compare(*std::prev(end), *child1))
96  {
97  if (child2 != end && compare(*child1, *child2))
98  {
99  *position = std::move(*child2);
100  position = child2;
101  }
102  else
103  {
104  *position = std::move(*child1);
105  position = child1;
106  }
107  }
108  else if (child2 != end && compare(*std::prev(end), *child2))
109  {
110  *position = std::move(*child2);
111  position = child2;
112  }
113  else
114  {
115  *position = std::move(*std::prev(end));
116  break;
117  }
118  }
119  }
