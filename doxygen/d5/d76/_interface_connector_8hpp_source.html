4 // The InterfaceConnector class is used to get access to a singleton class. It does not provide
5 // singleton functionality by itself, however. It is used by deriving your singleton class
6 // from InterfaceConnector, and upon instantiation of your singleton, InterfaceConnector registers
7 // where your singleton is. Via the static Instance function, everyone can access the singleton.
8 // Used together with an interface, this is a good way to decouple construction of your specific
9 // singleton from accessing the singleton via a generic interface. For example:
10 //
11 // class IDisplay
12 // : InterfaceConnector<IDisplay>
13 // {
14 // public:
15 // virtual void Write(char s) = 0;
16 // };
17 //
18 // --- In the HAL layer:
19 //
20 // class DisplayOnSTM
21 // : IDisplay
22 // {
23 // public:
24 // void Write(char s) { ... }
25 // };
26 //
27 // --- In an application layer, which is independent of the specific hardware implementation:
28 //
29 // void DoSomethingWithDisplay()
30 // {
31 // IDisplay::Instance().Write('x');
32 // }
33 //
34 // --- In main.cpp:
35 //
36 // int main()
37 // {
38 // static DisplayOnSTM display;
39 //
40 // ...
41 // }
