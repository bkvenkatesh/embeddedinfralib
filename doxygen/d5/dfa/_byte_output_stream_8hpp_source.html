4 // With a ByteOutputStreamWriter, you can easily place all sorts of objects into a block of memory.
5 // A ByteOutputStreamWriter is created with a MemoryRange as argument, objects streamed out of the
6 // ByteOutputStreamWriter are taken from that range.
7 //
8 // Example:
9 //
10 // uint16_t myData = 0x1234;
11 // uint16_t myChecksum = 0x5678;
12 //
13 // std::array<uint8_t, 20> memory;
14 // infra::ByteOutputStreamWriter writeStream(memory);
15 // writeStream << uint8_t(4) << myData << myChecksum;
16 //
17 // Now memory contains the bytes 0x04, 0x12, 0x34, 0x56, 0x78.
